# 📊 记录仪表盘

---

## 📅 日记打卡热力图

```dataviewjs
const calendarData = {
    year: new Date().getFullYear(),
    colors: {
        green: ["#c6e48b","#7bc96f","#49a845","#2e8840","#196127"],
    },
    showCurrentDayBorder: true,
    entries: [],
}

const pages = dv.pages('"10_日常/日记"').where(p => p.date)

for (let page of pages) {
    const d = page.date
    const dateStr = (d && d.toFormat) ? d.toFormat("yyyy-MM-dd") : String(d)
    calendarData.entries.push({
        date: dateStr,
        intensity: 3,
        content: page.file.name,
        color: "green",
    })
}

renderHeatmapCalendar(this.container, calendarData)
```

## 😊 心情热力图

```dataviewjs
const moodMap = {
    "很好": 5, "开心": 5, "愉快": 5,
    "好": 4, "不错": 4,
    "平静": 3, "一般": 3,
    "疲惫": 2, "低落": 2, "烦躁": 2,
    "很差": 1, "糟糕": 1,
}

const calendarData = {
    year: new Date().getFullYear(),
    colors: {
        mood: ["#fde8d8","#f9b99a","#f58a5e","#e85d2a","#c43a0f"],
    },
    showCurrentDayBorder: true,
    entries: [],
}

const pages = dv.pages('"10_日常/日记"').where(p => p.date && p.mood)

for (let page of pages) {
    const d = page.date
    const dateStr = (d && d.toFormat) ? d.toFormat("yyyy-MM-dd") : String(d)
    const moodText = String(page.mood)
    calendarData.entries.push({
        date: dateStr,
        intensity: moodMap[moodText] ?? 3,
        content: "心情：" + moodText,
        color: "mood",
    })
}

renderHeatmapCalendar(this.container, calendarData)
```
